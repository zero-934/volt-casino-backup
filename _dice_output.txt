/**
 * @file DiceDuelLogic.ts
 * @purpose Implements the core game logic for the Dice Duel game.
 * @author Agent 934
 * @date 2026-04-16
 * @license Proprietary
 */

import { ProvablyFairRNG } from '../shared/rng/ProvablyFairRNG';

export type DiceValue = 1 | 2 | 3 | 4 | 5 | 6;

export interface DiceDuelState {
  bet: number;
  playerDice: DiceValue[];      // [1-6, 1-6, 1-6]
  houseDice: DiceValue[];       // [1-6, 1-6, 1-6]
  playerTotal: number;
  houseTotal: number;
  doubledDown: boolean;      // true if player doubled their bet
  outcome: 'win' | 'lose' | 'push' | null;  // null = not resolved
  payout: number;            // Total amount returned to player (includes original bet)
  phase: 'bet' | 'player_rolled' | 'complete';  // game phase
  lastSeed: string;          // for provable fairness
}

export interface DiceDuelConfig {
  rng?: ProvablyFairRNG;
  houseEdge?: number; // Default 0.04 (4%)
}

const DICE_SIDES = 6;
const NUM_DICE = 3;
const HOUSE_EDGE_DEFAULT = 0.04;

/**
 * Creates the initial state for a new Dice Duel game.
 * @param bet The initial bet amount for the round.
 * @returns The initial DiceDuelState.
 */
export function createDiceDuelState(bet: number): DiceDuelState {
  if (bet <= 0) {
    throw new Error('Bet must be a positive number.');
  }
  return {
    bet,
    playerDice: [],
    houseDice: [],
    playerTotal: 0,
    houseTotal: 0,
    doubledDown: false,
    outcome: null,
    payout: 0,
    phase: 'bet',
    lastSeed: '',
  };
}

/**
 * Rolls the player's 3 dice and updates the game state.
 * Sets the phase to 'player_rolled'. The house dice are not rolled yet.
 * @param state The current DiceDuelState.
 * @param config Optional configuration including an RNG instance.
 * @returns The updated DiceDuelState.
 */
export function rollPlayerDice(state: DiceDuelState, config?: DiceDuelConfig): DiceDuelState {
  if (state.phase !== 'bet') {
    throw new Error('Cannot roll player dice outside of "bet" phase.');
  }

  const rng = config?.rng || new ProvablyFairRNG();
  const seed = rng.seed; // Store the seed for provable fairness

  const playerDice: DiceValue[] = [];
  let playerTotal = 0;
  for (let i = 0; i < NUM_DICE; i++) {
    const value = rng.randomInt(1, DICE_SIDES) as DiceValue;
    playerDice.push(value);
    playerTotal += value;
  }

  return {
    ...state,
    playerDice,
    playerTotal,
    phase: 'player_rolled',
    lastSeed: seed,
  };
}

/**
 * Doubles the player's bet. This action is only valid in the 'player_rolled' phase.
 * @param state The current DiceDuelState.
 * @returns The updated DiceDuelState with the doubled bet.
 */
export function doubleDown(state: DiceDuelState): DiceDuelState {
  if (state.phase !== 'player_rolled') {
    throw new Error('Cannot double down outside of "player_rolled" phase.');
  }
  if (state.doubledDown) {
    throw new Error('Cannot double down more than once.');
  }

  return {
    ...state,
    bet: state.bet * 2,
    doubledDown: true,
  };
}

/**
 * Rolls the house dice, resolves the round outcome, and calculates the payout.
 * Applies house edge by potentially flipping a player win to a house win.
 * Sets the phase to 'complete'.
 * @param state The current DiceDuelState.
 * @param config Optional configuration including an RNG instance and house edge.
 * @returns The updated DiceDuelState with the round resolved.
 */
export function resolveRound(state: DiceDuelState, config?: DiceDuelConfig): DiceDuelState {
  if (state.phase !== 'player_rolled') {
    throw new Error('Cannot resolve round outside of "player_rolled" phase.');
  }

  const rng = config?.rng || new ProvablyFairRNG();
  const houseEdge = config?.houseEdge ?? HOUSE_EDGE_DEFAULT;

  const houseDice: DiceValue[] = [];
  let houseTotal = 0;
  for (let i = 0; i < NUM_DICE; i++) {
    const value = rng.randomInt(1, DICE_SIDES) as DiceValue;
    houseDice.push(value);
    houseTotal += value;
  }

  let outcome: DiceDuelState['outcome'] = null;
  let payout = 0;

  if (state.playerTotal > houseTotal) {
    outcome = 'win';
    // Apply house edge: if player would win, check if RNG flips it to a loss
    if (rng.random() < houseEdge) {
      outcome = 'lose'; // House wins due to edge
    }
  } else if (state.playerTotal < houseTotal) {
    outcome = 'lose';
  } else {
    outcome = 'push';
  }

  switch (outcome) {
    case 'win':
      payout = state.bet * 2; // Player gets 2x their bet back (1x profit)
      break;
    case 'push':
      payout = state.bet; // Player gets their bet back
      break;
    case 'lose':
      payout = 0; // Player loses their bet
      break;
  }

  return {
    ...state,
    houseDice,
    houseTotal,
    outcome,
    payout,
    phase: 'complete',
  };
}

/**
 * Simulates a given number of Dice Duel rounds to estimate the Return to Player (RTP).
 * Assumes no double-down strategy for simplicity in simulation.
 * @param rounds The number of rounds to simulate.
 * @param bet The bet amount for each round.
 * @param config Optional configuration including an RNG instance and house edge.
 * @returns The estimated RTP as a fraction (e.g., 0.96 for 96%).
 */
export function simulateDiceDuelRTP(rounds: number, bet: number, config?: DiceDuelConfig): number {
  let totalWagered = 0;
  let totalPayout = 0;
  const rng = config?.rng || new ProvablyFairRNG();
  const simulationConfig = { ...config, rng }; // Ensure the same RNG instance is used across rolls in simulation

  for (let i = 0; i < rounds; i++) {
    let state = createDiceDuelState(bet);
    totalWagered += state.bet; // Initial bet

    state = rollPlayerDice(state, simulationConfig);
    // For simplicity in RTP simulation, we assume no double-down.
    // If double-down strategy were to be simulated, it would be applied here.

    state = resolveRound(state, simulationConfig);
    totalPayout += state.payout;
  }

  return totalPayout / totalWagered;
}

  /**

   * @param container The container for the dice (playerDiceContainer or houseDiceContainer).
   * @param values An array of dice values (1-6).
   * @param isHouse True if updating house dice, false for player dice.
   * @param isFaceDown True if dice should be shown face down.

  public updateDiceDisplay(container: Phaser.GameObjects.Container, values: DiceValue[], isHouse: boolean, isFaceDown: boolean): void {
    container.removeAll(true); // Clear existing dice

    const startX = GAME_WIDTH / 2 - (DICE_SIZE * 3 + DICE_SPACING * 2) / 2 + DICE_SIZE / 2;
    const y = isHouse ? HOUSE_DICE_Y : PLAYER_DICE_Y;

  /**
   * Draws a single die graphic.
   * @param x X-coordinate of the die center.
   * @param y Y-coordinate of the die center.
   * @param value The dice value (1-6) or null for face-down.
   * @param isFaceDown True if the die should be face down.
   * @returns A Phaser.GameObjects.Graphics object representing the die.
   */
  private drawDie(x: number, y: number, value: DiceValue | null, isFaceDown: boolean): Phaser.GameObjects.Graphics {
    const graphics = this.scene.add.graphics({ x: x - DICE_SIZE / 2, y: y - DICE_SIZE / 2 });

    if (isFaceDown) {
      graphics.fillStyle(DARK, 1);
      graphics.fillRoundedRect(0, 0, DICE_SIZE, DICE_SIZE, DICE_CORNER_RADIUS);
      graphics.lineStyle(DICE_BORDER_WIDTH, GOLD, 1);
      graphics.strokeRoundedRect(0, 0, DICE_SIZE, DICE_SIZE, DICE_CORNER_RADIUS);

      const questionMark = this.scene.add.text(x, y, '?', {
        fontFamily: FONT_FAMILY,
        fontSize: '48px',
        color: GOLD_STR,
        align: 'center'
      }).setOrigin(0.5);
      this.container.add(questionMark);
      // Attach question mark to graphics for easier management if needed, or just add to container
      // For simplicity, we'll just return the graphics and let the scene manage the text separately if needed.
      // Or, better, create a container for each die. Let's adjust this.
      const dieContainer = this.scene.add.container(x, y);
      dieContainer.add(graphics);
      dieContainer.add(questionMark);
      return graphics; // Returning graphics for now, but a container would be better for complex dice.
    }

    // Draw shadow/depth effect
    graphics.fillStyle(0xaaaaaa, 1); // Light grey for depth
    graphics.fillRoundedRect(DICE_BORDER_WIDTH / 2, DICE_BORDER_WIDTH / 2 + 2, DICE_SIZE - DICE_BORDER_WIDTH, DICE_SIZE - DICE_BORDER_WIDTH, DICE_CORNER_RADIUS);

    // Draw main white body
    graphics.fillStyle(0xffffff, 1);
    graphics.fillRoundedRect(DICE_BORDER_WIDTH / 2, DICE_BORDER_WIDTH / 2, DICE_SIZE - DICE_BORDER_WIDTH, DICE_SIZE - DICE_BORDER_WIDTH, DICE_CORNER_RADIUS);

    // Draw gold border
    graphics.lineStyle(DICE_BORDER_WIDTH, GOLD, 1);
    graphics.strokeRoundedRect(DICE_BORDER_WIDTH / 2, DICE_BORDER_WIDTH / 2, DICE_SIZE - DICE_BORDER_WIDTH, DICE_SIZE - DICE_BORDER_WIDTH, DICE_CORNER_RADIUS);

    // Draw dots
    graphics.fillStyle(DARK, 1);
    const dotPositions = this.getDotPositions(value!);
    dotPositions.forEach(pos => {
      graphics.fillCircle(DICE_SIZE / 2 + pos.x, DICE_SIZE / 2 + pos.y, DICE_DOT_RADIUS);
    });

    return graphics;
  }

  /**
   * Calculates the relative positions of dots for a given dice value.
   * @param value The dice value (1-6).
   * @returns An array of {x, y} coordinates relative to the die center.
   */
  private getDotPositions(value: DiceValue): { x: number; y: number }[] {
    const offset = DICE_SIZE / 4; // Offset from center for dots
    const positions: { x: number; y: number }[] = [];

    const addDot = (x: number, y: number) => positions.push({ x, y });

    if (value === 1 || value === 3 || value === 5) {
      addDot(0, 0); // Center dot
    }
    if (value === 2 || value === 3 || value === 4 || value === 5 || value === 6) {
      addDot(-offset, -offset); // Top-left
      addDot(offset, offset);   // Bottom-right
    }
    if (value === 4 || value === 5 || value === 6) {
      addDot(-offset, offset);  // Bottom-left
      addDot(offset, -offset);  // Top-right
    }
    if (value === 6) {
      addDot(-offset, 0);       // Mid-left
      addDot(offset, 0);        // Mid-right
    }
    return positions;
  }

        resultText = `YOU WIN! +$${profit}`;
        textColor = GREEN;
        break;
      case 'lose':

        resultText = `YOU LOSE! -$${doubledDown ? (payout / 2) : (payout / 1)}`;
        resultText = `YOU LOSE!`; // Actual loss is handled by scene balance update
        textColor = RED;
        break;
      case 'push':

        textColor = GOLD_STR;
        break;

/**

   * Handles the player clicking the ROLL button.
   */
  private handleRoll(): void {
    if (this.state.phase !== 'bet') return;

    if (this.balance < this.currentBet) {
      // Not enough balance, show message
      this.diceDuelUI.showResult('lose', 0, false); // Use result text to show error
      this.resultText.setText('NOT ENOUGH BALANCE!').setColor(RED).setVisible(true);
      this.time.delayedCall(2000, () => this.resultText.setVisible(false));
      return;
    }

    this.balance -= this.currentBet;
    this.diceDuelUI.updateBalanceDisplay(this.balance);
    this.diceDuelUI.updateBetDisplay(this.currentBet); // Update bet display in case it was changed

    this.state = rollPlayerDice(this.state, { rng: this.rng });
    this.diceDuelUI.updateDiceDisplay(this.diceDuelUI['playerDiceContainer'], this.state.playerDice, false, false);
    this.diceDuelUI.playerTotalText.setText(`YOUR ROLL: ${this.state.playerTotal}`);
    this.diceDuelUI.showDoubleDownButton();

    // Start auto-resolve timer
    this.autoResolveTimer = this.time.delayedCall(AUTO_RESOLVE_DELAY_MS, this.handleResolve, [], this);
  }

   * Handles the player clicking the DOUBLE DOWN button.
   */
  private handleDoubleDown(): void {
    if (this.state.phase !== 'player_rolled' || this.state.doubledDown) return;

    if (this.balance < this.currentBet) { // Check if player has enough balance for the *additional* bet
      // Not enough balance for double down
      this.diceDuelUI.showResult('lose', 0, false);
      this.resultText.setText('NOT ENOUGH BALANCE TO DOUBLE DOWN!').setColor(RED).setVisible(true);
      this.time.delayedCall(2000, () => this.resultText.setVisible(false));
      return;
    }

    this.balance -= this.currentBet; // Deduct the additional bet
    this.diceDuelUI.updateBalanceDisplay(this.balance);

    this.state = doubleDown(this.state);
    this.diceDuelUI.updateBetDisplay(this.state.bet); // Update UI to show doubled bet
    this.diceDuelUI.hideDoubleDownButton();

    // If auto-resolve timer is active, let it continue, otherwise start it (should already be active)
    if (!this.autoResolveTimer) {
      this.autoResolveTimer = this.time.delayedCall(AUTO_RESOLVE_DELAY_MS, this.handleResolve, [], this);
    }
  }

  /**
   * Handles resolving the round (rolling house dice and determining outcome).
   */
  private handleResolve(): void {
    if (this.autoResolveTimer) {
      this.autoResolveTimer.destroy();
      this.autoResolveTimer = null;
    }
    if (this.state.phase !== 'player_rolled') return;

    this.diceDuelUI.hideDoubleDownButton(); // Hide double down button if still visible

    this.state = resolveRound(this.state, { rng: this.rng });

    // Animate house dice reveal
    this.time.delayedCall(500, () => {
      this.diceDuelUI.updateDiceDisplay(this.diceDuelUI['houseDiceContainer'], this.state.houseDice, true, false);
      this.diceDuelUI.houseTotalText.setText(`HOUSE ROLL: ${this.state.houseTotal}`);

      // Update balance and show result
      this.balance += this.state.payout;
      this.diceDuelUI.updateBalanceDisplay(this.balance);
      this.diceDuelUI.showResult(this.state.outcome, this.state.payout, this.state.doubledDown);

      // Animate dice based on outcome
      if (this.state.outcome === 'win') {
        this.diceDuelUI.animateDiceResult('win', this.diceDuelUI['playerDiceContainer']);
      } else if (this.state.outcome === 'lose') {
        this.diceDuelUI.animateDiceResult('lose', this.diceDuelUI['playerDiceContainer']);
      }

      this.diceDuelUI.showPlayAgainButton();
    });
  }

  /**
   * Updates the UI elements based on the current game state.
   */
  private updateUIFromState(): void {
    this.diceDuelUI.updateBetDisplay(this.state.bet);
    this.diceDuelUI.updateBalanceDisplay(this.balance);
    this.diceDuelUI.updateDiceDisplay(this.diceDuelUI['playerDiceContainer'], this.state.playerDice, false, this.state.phase === 'bet');
    this.diceDuelUI.updateDiceDisplay(this.diceDuelUI['houseDiceContainer'], this.state.houseDice, true, this.state.phase !== 'complete');
    this.diceDuelUI.updateTotals(this.state.playerTotal, this.state.houseTotal);

    if (this.state.phase === 'bet') {
      this.diceDuelUI.showRollButton();
    } else if (this.state.phase === 'player_rolled') {
      this.diceDuelUI.showDoubleDownButton();
    } else if (this.state.phase === 'complete') {
      this.diceDuelUI.showResult(this.state.outcome, this.state.payout, this.state.doubledDown);
      this.diceDuelUI.showPlayAgainButton();
    }
  }

  /**
   * Restarts the game by resetting the scene.
   */
  private restartGame(): void {
    this.scene.restart();
  }
}

    expect(state.houseDice).toEqual([]);
    expect(state.houseTotal).toBe(0);
  });

  // 4. doubleDown doubles the bet and sets doubledDown=true
  it('4. doubleDown doubles the bet and sets doubledDown=true', () => {
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled', // Must be in this phase
      playerDice: [1, 1, 1],
      playerTotal: 3,
    };
    state = doubleDown(state);

    expect(state.bet).toBe(INITIAL_BET * 2);
    expect(state.doubledDown).toBe(true);
  });

  // 5. doubleDown only works in phase='player_rolled'
  it('5. doubleDown only works in phase="player_rolled"', () => {
    let state = createDiceDuelState(INITIAL_BET); // phase: 'bet'
    expect(() => doubleDown(state)).toThrow('Cannot double down outside of "player_rolled" phase.');

    state.phase = 'complete';
    expect(() => doubleDown(state)).toThrow('Cannot double down outside of "player_rolled" phase.');
  });

  // 6. resolveRound fills houseDice with 3 values 1-6 and sets phase='complete'
  it('6. resolveRound fills houseDice with 3 values 1-6 and sets phase="complete"', () => {

      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',

    expect(state.houseTotal).toBe(6);
    expect(state.phase).toBe('complete');
  });

  // 7. resolveRound: player wins when playerTotal > houseTotal
  it('7. resolveRound: player wins when playerTotal > houseTotal', () => {
    const mockRNG = new MockRNG([0.1, 0.1, 0.1, 0.0]); // House dice: 1,1,1. House edge check: 0.0 (no flip)
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',
      playerDice: [6, 6, 6],
      playerTotal: 18,

    state = resolveRound(state, { rng: mockRNG, houseEdge: 0.1 }); // Player 18, House 3

    expect(state.outcome).toBe('win');
    expect(state.payout).toBe(INITIAL_BET * 2);
  });

  // 8. resolveRound: push when totals equal
  it('8. resolveRound: push when totals equal', () => {

  // 9. resolveRound: house wins when houseTotal > playerTotal
  it('9. resolveRound: house wins when houseTotal > playerTotal', () => {
    const mockRNG = new MockRNG([0.6, 0.6, 0.6]); // House dice: 4,4,4
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',
      playerDice: [1, 1, 1],
      playerTotal: 3,
    };
    state = resolveRound(state, { rng: mockRNG, houseEdge: 0 }); // Player 3, House 12

    expect(state.payout).toBe(0);
  });

  // 10. payout correct on normal win (bet*2 - bet = bet profit, so payout = bet*2)
  it('10. payout correct on normal win (payout = bet * 2)', () => {
    const mockRNG = new MockRNG([0.1, 0.1, 0.1, 0.0]); // House dice: 1,1,1. House edge check: 0.0 (no flip)
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',
      playerDice: [6, 6, 6],

    expect(state.outcome).toBe('win');
    expect(state.payout).toBe(INITIAL_BET * 2); // Payout is 2x original bet
  });

  // 11. payout correct on double-down win (bet*4 - bet*2 = bet*2 profit, so payout = bet*4)
  it('11. payout correct on double-down win (payout = bet * 4)', () => {
    const mockRNG = new MockRNG([0.1, 0.1, 0.1, 0.0]); // House dice: 1,1,1. House edge check: 0.0 (no flip)
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',

    };

    expect(state.outcome).toBe('win');
    expect(state.payout).toBe(INITIAL_BET * 4); // Payout is 2x the doubled bet
  });

  // Test house edge flipping a win to a loss
  it('11b. resolveRound: house edge can flip a player win to a loss', () => {
    // Player rolls 18, House rolls 3. Player should win.
    // RNG for house dice: [0.1, 0.1, 0.1] -> 1,1,1 (total 3)
    // RNG for house edge check: [0.03] -> less than 0.04 house edge, so flip to lose
    const mockRNG = new MockRNG([0.1, 0.1, 0.1, 0.03]);
    let state: DiceDuelState = {
      ...createDiceDuelState(INITIAL_BET),
      phase: 'player_rolled',
      playerDice: [6, 6, 6],
      playerTotal: 18,
    };
    state = resolveRound(state, { rng: mockRNG, houseEdge: 0.04 });

    expect(state.outcome).toBe('lose');
    expect(state.payout).toBe(0);
  });

  // 12. simulateDiceDuelRTP over 5000 rounds returns between 0.70 and 1.10
